cmake_minimum_required(VERSION 3.30)
project(easily-bootstrappable-pdc
  VERSION 1.0.0
  DESCRIPTION "Distribución de PseudoD minimalista"
  HOMEPAGE_URL "https://github.com/alinarezrangel/pseudod-v3-pdc"
  LANGUAGES C
)

include(ExternalProject)
include(GNUInstallDirs)

set(boot_t1 "${CMAKE_CURRENT_BINARY_DIR}/boot-t1")
file(MAKE_DIRECTORY "${boot_t1}")
ExternalProject_Add(boot_t1
        SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/boot-t1"
        BINARY_DIR "${boot_t1}"
        PREFIX "${CMAKE_CURRENT_BINARY_DIR}/boot_t1-prefix"
        INSTALL_DIR ${CMAKE_INSTALL_PREFIX}
        CMAKE_ARGS
            -DCMAKE_INSTALL_PREFIX:PATH=<INSTALL_DIR>
            -DCMAKE_BUILD_TYPE:STRING=${CMAKE_BUILD_TYPE}
        CMAKE_CACHE_ARGS
            -DBOOT_PREFIX:STRING=bt1-

            -DCMAKE_TOOLCHAIN_FILE:STRING=${CMAKE_TOOLCHAIN_FILE}
            -DCMAKE_C_COMPILER:STRING=${CMAKE_C_COMPILER}
            -DCMAKE_CXX_COMPILER:STRING=${CMAKE_CXX_COMPILER}

            -DCMAKE_INSTALL_BINDIR:STRING=${CMAKE_INSTALL_BINDIR}
            -DCMAKE_INSTALL_SBINDIR:STRING=${CMAKE_INSTALL_SBINDIR}
            -DCMAKE_INSTALL_LIBEXECDIR:STRING=${CMAKE_INSTALL_LIBEXECDIR}
            -DCMAKE_INSTALL_SYSCONFDIR:STRING=${CMAKE_INSTALL_SYSCONFDIR}
            -DCMAKE_INSTALL_SHAREDSTATEDIR:STRING=${CMAKE_INSTALL_SHAREDSTATEDIR}
            -DCMAKE_INSTALL_LOCALSTATEDIR:STRING=${CMAKE_INSTALL_LOCALSTATEDIR}
            -DCMAKE_INSTALL_RUNSTATEDIR:STRING=${CMAKE_INSTALL_RUNSTATEDIR}
            -DCMAKE_INSTALL_LIBDIR:STRING=${CMAKE_INSTALL_LIBDIR}
            -DCMAKE_INSTALL_INCLUDEDIR:STRING=${CMAKE_INSTALL_INCLUDEDIR}
            -DCMAKE_INSTALL_OLDINCLUDEDIR:STRING=${CMAKE_INSTALL_OLDINCLUDEDIR}
            -DCMAKE_INSTALL_DATAROOTDIR:STRING=${CMAKE_INSTALL_DATAROOTDIR}
            -DCMAKE_INSTALL_DATADIR:STRING=${CMAKE_INSTALL_DATADIR}
            -DCMAKE_INSTALL_INFODIR:STRING=${CMAKE_INSTALL_INFODIR}
            -DCMAKE_INSTALL_LOCALEDIR:STRING=${CMAKE_INSTALL_LOCALEDIR}
            -DCMAKE_INSTALL_MANDIR:STRING=${CMAKE_INSTALL_MANDIR}
            -DCMAKE_INSTALL_DOCDIR:STRING=${CMAKE_INSTALL_DOCDIR}
)


add_custom_target(latest DEPENDS boot_t1)
